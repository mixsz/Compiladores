expressao --> soma

soma --> mult somaResto

somaResto --> '+' mult somaResto | '-' mult somaResto | ε

mult --> valor multResto

multResto --> '*' valor multResto | '/' valor multResto | ε

valor --> inteiro | decimal | id | '(' expressao ')' | '-' valor | texto




tipos --> tipoInt | tipoFloat | tipoString

declarar --> tipos id inicializar

inicializar --> opAtrib conteudoInicial | ε

conteudoInicial --> expressao | leia '(' tipos ')'

atribuir --> id opAtrib destinoAtribuicao

destinoAtribuicao --> expressao | leia '(' tipos ')'

comentar --> comentario texto

crementar --> id opCremento



estruturaIf --> se '(' condicao ')' '{' bloco '}' estruturaElseif estruturaElse

estruturaElseif --> senaose '(' condicao ')' '{' bloco '}' estruturaElseif | ε

estruturaElse --> senao '{' bloco '}' | ε





condicao --> termoE condicaoResto

condicaoResto --> ou termoE condicaoResto | ε

termoE --> relacional termoEResto

termoEResto --> e relacional termoEResto | ε

relacional --> expressao opRelac expressao | '(' condicao ')'




estruturaWhile --> enquanto '(' condicao ')' '{' bloco '}'

estruturaFor --> para '(' (atribuir|declarar) fim condicao fim (atribuir|crementar) ')' '{' bloco '}'




estruturaEscrever --> escreva '('  (expressao) restoTexto ')'

restoTexto --> '+' (expressao) restoTexto | ε




iniciar --> inicie ':' bloco




bloco --> instrucao bloco | ε

instrucao --> declarar fim | atribuir fim | crementar fim | estruturaIf | estruturaWhile | estruturaFor | quebre fim | continue fim | estruturaEscrever fim | comentar fim
